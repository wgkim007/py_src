			var url = "/main.do";

					var mbrid = search.replace("noticeinfo", "");
					url = "/board/view.do?board_name=normal_notice&mbr_id=" + mbrid;
					$("#mainifr").attr("src", url);
				} else if(search.indexOf("eventinfo") == 0) {
					var mbrid = search.replace("eventinfo", "");
					url = "/board/view.do?board_name=event_board&mbr_id=" + mbrid;
					$("#mainifr").attr("src", url);
				}
				else if(search.indexOf("fundinfo") == 0) {
					var fundcode = search.replace("fundinfo", "");
					url = "/product/fund/detail/ol5902q_1_detail.to?scr_menu_id=PD&fund_cd=" + fundcode;
					$("#mainifr").attr("src", url);
				}
				else if(search.indexOf("branchinfo") == 0) {
					var branchcode = search.replace("branchinfo", "");
					url = "/company/co_info/network/ol0001qQ_101_2.to?br_cd=" + branchcode;
					$("#mainifr").attr("src", url);
				}
				else {
					var idx = search.indexOf("&");
					var menu_id = "";
					var param = "";
					if(idx > 0) {
						menu_id = search.substring(0, idx);

		function initialize(){
			
			// IE memory leaking 방지
			if ( window.attachEvent && !window.addEventListener ) {
			    window.attachEvent("onunload", function() {
			        for ( var id in jQuery.cache ) {
			            if ( jQuery.cache[ id ].handle ) {
			                // Try/Catch is to handle iframes being unloaded, see #4280
			                try {
			                    jQuery.event.remove( jQuery.cache[ id ].handle.elem   );
			                } catch(e) {}
			            }
			        }
			    });
			}
			
			callPage();
		}
